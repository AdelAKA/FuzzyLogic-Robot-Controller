    # input :
        # L err,angle = [N, SN, NNZ, Z, NPZ, SP, P]
        # L distance = [Z, NZ, NM, M, NF, F, VF]
    # output :
        # L left/right velocity = [Z, S, NM, M, NH, H, VH]
# OAFLC -> avoid obsticle
    # input :
        # L left/right/front obs distance = [N, M, F] 
    # output:
        # L left/right velocity = [NH, N, P, HP, VHP]
